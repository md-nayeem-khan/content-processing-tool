    def __init__(self):
        self.result = 0

    def add(self, value):
        self.result += value
        return self.result

    def subtract(self, value):
        self.result -= value
        return self.result

function processData(items) {

        .map(item => ({
            id: item.id,
            name: item.name.toUpperCase()
        }))
        .sort((a, b) => a.name.localeCompare(b.name));
